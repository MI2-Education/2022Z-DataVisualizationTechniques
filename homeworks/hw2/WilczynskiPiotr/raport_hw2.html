# necessary libraries
library(dplyr)
library(ggplot2)
library(ggrepel)

# Europe
df_europe <- read.csv("crim_off_cat_1_Data.csv")

df_europe <- df_europe %>% 
  filter(TIME >= 2014 &
          ICCS == "Intentional homicide" &
           UNIT == "Number" &
            Value != ":")

df_europe$Value <- as.integer(df_europe$Value)

df_europe <- df_europe %>% 
  group_by(TIME) %>% 
  summarize(homicides_number = sum(Value))

# UK 2014 - 2019
uk_homicides <- c(601, 611, 641, 728, 774, 650)
df_europe$homicides_number <- df_europe$homicides_number + uk_homicides

# US 2014 - 2019
df_US <- read.csv("US_homicide.csv")

# Merge
df_europe <- df_europe %>% 
  mutate(region = "Europe")
df_US <- df_US %>% 
  mutate(region = "United States")

df <- rbind(df_europe, df_US)

# Plot
ggplot(df, aes(x = TIME, y = homicides_number, group = region, color = region)) +
  geom_line(size = 1) +
  scale_y_continuous(limits = c(0, 20000)) +
  scale_x_continuous() +
  labs(
    title = "Number of Homicides in the USA and Europe (2014-2019)",
    x = "Year",
    y = "Number of Homicides"
  ) +
  scale_color_manual(values = c("blue", "red")) +
  geom_label_repel(label = df$homicides_number, min.segment.length = unit(0, 'lines'),
                   nudge_y = 1500) +
  geom_text(aes(x = 2014.5, y = 4500, label = "Europe", color = "Europe")) + 
  geom_text(aes(x = 2014.5, y = 13000, label = "United States", color = "United States")) +
  geom_point() +
  theme_bw() +
  theme(
    legend.position = "None",
    panel.grid.major =  element_line(size = 0.5),
    panel.grid.minor.x = element_blank(),
    axis.ticks = element_line(size = 2)
  ) +
  expand_limits(x = c(2013.9, 2019.1))
